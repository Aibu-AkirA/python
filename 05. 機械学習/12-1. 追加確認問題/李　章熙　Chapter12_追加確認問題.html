以下の条件で折れ線グラフを作成するプログラムを完成させなさい。
（参考：12.1.1 マーカーの種類と色を設定する、12.1.2 線のスタイルと色を設定する）

  • y軸のデータは [-1, 0, 4, 12, 17, 22, 25, 26, 22, 16, 9, 2] とする

import numpy as np
%matplotlib inline

# Y軸データ(問題文の値)

# グラフ表示
plt.show()

import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline

np.random.seed(0)

# x軸、y軸、z軸データを作成
x = np.random.normal(size=2000)
y = np.random.normal(size=2000)

以下の条件で円グラフを作成するプログラムを完成させなさい。
（参考：12.5.1 円グラフを作成する～12.5.3 特定の要素を目立たせる）

  • データは [ 24, 23, 19, 8, 7, 4, 15 ] とする
  • データに対応するラベルは [ "Korea", "China", "Taiwan", "Thailand", "HongKong", "Malaysia", "Other" ] とする
  • データの目立たせ度合いについて、"Taiwan"のデータを0.2、その他は0とする

import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline

# 円グラフデータの設定
data = [24, 23, 19, 8, 7, 4, 15]

# データに対応するラベルの設定
labels = ["Korea", "China", "Taiwan", "Thailand", "HongKong", "Malaysia", "Other"]

# データの目立たせ度合いの設定
explode = [0, 0, 0.2, 0, 0, 0, 0]

# 円グラフの作成
plt.pie(data, labels=labels, explode=explode)

# 真円になるようaxisの設定
plt.axis("equal")

plt.show()

import numpy as np
import matplotlib.pyplot as plt
# 3D描画を行うために必要なライブラリのimport文を追加
from mpl_toolkits.mplot3d import Axes3D
# カラーマップを表示するためのライブラリのimport文を追加
from matplotlib import cm

%matplotlib inline

# X軸データ
x = np.linspace(-1, 1, 50)
# Y軸データ
y = np.linspace(-1, 1, 50)

# X軸・Y軸データを組み合わせた格子状のデータを作成
X, Y = np.meshgrid(x,y)

# Z軸データ
Z = np.cos(X)

# 大きさが縦横とも6インチのFigureオブジェクトを作成
fig = plt.figure(figsize=(6, 6))

# 3D描画できるサブプロットaxを作成
ax = fig.add_subplot(1, 1, 1, projection="3d")

# 曲面をプロット
# Zの値についてカラーマップを適用
ax.plot_surface(X, Y, Z)

plt.show()

import matplotlib.pyplot as plt
import numpy as np
# 3D描画を行うために必要なライブラリのimport文を追加
from mpl_toolkits.mplot3d import Axes3D

%matplotlib inline

# 大きさが縦横とも5インチのFigureオブジェクトを作成

# 3D描画できるサブプロットaxを作成
ax = fig.add_subplot(1, 1, 1, projection="3d")

ypos = list(range(9,-1,-1))
